## Installation steps
- pkg upgrade -y && pkg install bash -y && pkg install shc -y && pkg install binutils -y && termux-setup-storage
- cd /sdcard/ZmodXv3.0 && sh rish
- ls
- sh setup.sh
- cd /data/local/tmp
- ./ZMODX
